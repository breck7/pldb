(define (bottles n more)
  (printf &quot;~a bottle~a of beer~a&quot;
          (case n [(0) &quot;no more&quot;] [(1) &quot;1&quot;] [else n])
          (if (= n 1) &quot;&quot; &quot;s&quot;)
          more))

(for ([n (in-range 99 0 -1)])
  (bottles n &quot; on the wall, &quot;)
  (bottles n &quot;.\n&quot;)
  (printf &quot;Take one down and pass it around, &quot;)
  (bottles (sub1 n) &quot; on the wall.\n\n&quot;))

  (cond [(zero? n) 1]
        [else (* n (fact (- n 1)))]))
