func gcd( int left right ) int
{
    if right == 0 : return left
    return gcd( right, left % right )
}

run  {
    str     input
    int     left right
    Println(&quot;This program finds the greatest common divisor by the Euclidean Algorithm.&quot;)
 
    while true
    {
       left = int( ReadString( &quot;Enter the first number ( enter 0 to exit ): &quot;))
       if left == 0 : break
       
       right = int( ReadString( &quot;Enter the second number: &quot;))
       Println(&quot;GCD = \{ gcd( left, right )}&quot;)
    }
}
