use strict;
use warnings;
use IO::Handle;

my ( $remaining, $total );

$remaining = $total = shift(@ARGV);

STDOUT-&gt;autoflush(1);

while ( $remaining ) {
    printf ( &quot;Remaining %s/%s \r&quot;, $remaining--, $total );
    sleep 1;
}

print &quot;\n&quot;;
