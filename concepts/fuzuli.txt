(require &quot;/usr/lib/fuzuli/nfl/math.nfl&quot;)
(function euclidean (params x y)
    (block
        (def dist FLOAT)(let dist 0.0)
        (def i INTEGER)
        (for (let i 0) (&lt; i (length x)) (inc i)
            (let dist (+ dist (pow (- (nth x i) (nth y i)) 2)))
        )
    (return dist)
    )
)

(def y LIST)
