OpenOffice Basic (formerly known as StarOffice Basic or StarBasic or OOoBasic) is a dialect of the programming language BASIC that originated with the StarOffice office suite and spread through OpenOffice.org and derivatives such as LibreOffice (where it is known as LibreOffice Basic). Sub ParaCount
&#39;
&#39; Count number of paragraphs in a text document
&#39;
    Dim Doc As Object, Enum As Object, TextEl As Object, Count As Long
    Doc = ThisComponent
&#39; Is this a text document?
    If Not Doc.SupportsService(&quot;com.sun.star.text.TextDocument&quot;) Then
        MsgBox &quot;This macro must be run from a text document&quot;, 64, &quot;Error&quot;
        Exit Sub
    End If
    Count = 0
&#39; Examine each component - paragraph or table?
    Enum = Doc.Text.CreateEnumeration
    While Enum.HasMoreElements
        TextEl = Enum.NextElement
&#39; Is the component a paragraph?
        If TextEl.SupportsService(&quot;com.sun.star.text.Paragraph&quot;) Then
            Count = Count + 1
        End If
    Wend
&#39;Display result
    MsgBox Count, 0, &quot;Paragraph Count&quot;
End Sub
