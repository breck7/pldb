def pulls(strength real):
    normal(strength, 1)

def winner(a real, b real):
    a_pull &lt;~ pulls(a)
    b_pull &lt;~ pulls(b)
    return (a_pull &gt; b_pull)

alice &lt;~ normal(0,1)
