Print(&quot;Hello, world!\n&quot;);

gap&gt; START_TEST(&quot;Test of factor groups and natural homomorphisms&quot;);

gap&gt; G:=HeisenbergPcpGroup(2);
Pcp-group with orders [ 0, 0, 0, 0, 0 ]

gap&gt; H:=Subgroup(G,[G.2,G.3,G.4,G.5]);
gap&gt; K:=G/H;
gap&gt; NaturalHomomorphism(K);

gap&gt; A:=Subgroup(H, [G.3]);
Pcp-group with orders [ 0 ]
gap&gt; B:=Subgroup(Subgroup(G,[G.1,G.4,G.5]), [G.4]);
Pcp-group with orders [ 0 ]
gap&gt; Normalizer(A,B);
Pcp-group with orders [ 0 ]
gap&gt; # The following used to trigger the error &quot;arguments must have a common parent group&quot;
gap&gt; Normalizer(B,A);
Pcp-group with orders [ 0 ]

gap&gt; STOP_TEST( &quot;factor.tst&quot;, 10000000);

gap&gt; G:=SmallGroup(8,1); # Set G to be a group of order 8.
&lt;pc group of size 8 with 3 generators&gt;
gap&gt; i:=IsomorphismPermGroup(G); # Find an isomorphism from G to a group of permutations
&lt;action isomorphism&gt;
gap&gt; Image(i,G); # The image of G under I - these are the generators of im G.
Group([ (1,5,3,7,2,6,4,8), (1,3,2,4)(5,7,6,8), (1,2)(3,4)(5,6)(7,8) ])
gap&gt; Elements(Image(i,G)); # All the elements of im G.
[ (), (1,2)(3,4)(5,6)(7,8), (1,3,2,4)(5,7,6,8), (1,4,2,3)(5,8,6,7),
