class App.FromNowView extends Ember.View
    tagName: &#39;time&#39;
    template: Ember.Handlebars.compile &#39;{{view.output}}&#39;
    output: ~&gt;
        return moment(@value).fromNow()

    didInsertElement: -&gt;
        @tick()

    tick: -&gt;
        f = -&gt;
            @notifyPropertyChange &#39;output&#39;
            @tick()

        nextTick = Ember.run.later(this, f, 1000)
        @set &#39;nextTick&#39;, nextTick

    willDestroyElement: -&gt;
        nextTick = @nextTick
        Ember.run.cancel nextTick

Ember.Handlebars.helper &#39;fromNow&#39;, App.FromNowView
