/**
 * Question 1.11
 * vabs(?Val, ?AbsVal)
 */
vabs(Val, AbsVal):-
	AbsVal #&gt; 0,
	(
		Val #= AbsVal
	;
		Val #= -AbsVal
	),
	labeling([Val, AbsVal]).

/**
 * vabsIC(?Val, ?AbsVal)
 */
vabsIC(Val, AbsVal):-
	AbsVal #&gt; 0,
	Val #= AbsVal or Val #= -AbsVal,
	labeling([Val, AbsVal]).

/**
 * Question 1.13
 * faitListe(?ListVar, ?Taille, +Min, +Max)
 */
faitListe([], 0, _, _):-!.
faitListe([First|Rest], Taille, Min, Max):-
	First #:: Min..Max,
	Taille1 #= Taille - 1,
	faitListe(Rest, Taille1, Min, Max).

 * Question 1.14
 * suite(?ListVar)
 */
suite([Xi, Xi1, Xi2]):-
	checkRelation(Xi, Xi1, Xi2).
suite([Xi, Xi1, Xi2|Rest]):-
	checkRelation(Xi, Xi1, Xi2),
	suite([Xi1, Xi2|Rest]).

/**
 * checkRelation(?Xi, ?Xi1, ?Xi2)
 */
checkRelation(Xi, Xi1, Xi2):-
	vabs(Xi1, VabsXi1),
	Xi2 #= VabsXi1 - Xi.
