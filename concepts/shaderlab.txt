// From https://github.com/Unity-Technologies/PostProcessing,
// licensed under MIT licence.

Shader &quot;Hidden/Post FX/Depth Of Field&quot;
{
    Properties
    {
        _MainTex (&quot;&quot;, 2D) = &quot;black&quot;
    }

    CGINCLUDE
        #pragma exclude_renderers d3d11_9x
        #pragma target 3.0
    ENDCG

    SubShader
    {
        Cull Off ZWrite Off ZTest Always

        // (0) Downsampling, prefiltering &amp; CoC
        Pass
        {
            CGPROGRAM
                #pragma multi_compile __ UNITY_COLORSPACE_GAMMA
                #pragma vertex VertDOF
                #pragma fragment FragPrefilter
                #include &quot;DepthOfField.cginc&quot;
            ENDCG
        }

        // (1) Pass 0 + temporal antialiasing
        Pass
        {
            CGPROGRAM
                #pragma vertex VertDOF
                #pragma fragment FragPrefilter
                #define PREFILTER_TAA
                #include &quot;DepthOfField.cginc&quot;
            ENDCG
        }

        // (2-5) Bokeh filter with disk-shaped kernels
        Pass
        {
            CGPROGRAM
                #pragma vertex VertDOF
                #pragma fragment FragBlur
                #define KERNEL_SMALL
                #include &quot;DepthOfField.cginc&quot;
            ENDCG
        }

        Pass
        {
            CGPROGRAM
                #pragma vertex VertDOF
                #pragma fragment FragBlur
                #define KERNEL_MEDIUM
                #include &quot;DepthOfField.cginc&quot;
            ENDCG
        }

        Pass
        {
            CGPROGRAM
                #pragma vertex VertDOF
                #pragma fragment FragBlur
                #define KERNEL_LARGE
                #include &quot;DepthOfField.cginc&quot;
            ENDCG
        }

        Pass
        {
            CGPROGRAM
                #pragma vertex VertDOF
                #pragma fragment FragBlur
                #define KERNEL_VERYLARGE
                #include &quot;DepthOfField.cginc&quot;
            ENDCG
        }

        // (6) Postfilter blur
        Pass
        {
            CGPROGRAM
                #pragma vertex VertDOF
                #pragma fragment FragPostBlur
                #include &quot;DepthOfField.cginc&quot;
            ENDCG
        }
    }

    FallBack Off
}
