{-# LANGUAGE EmptyDataDecls #-}
module Hello where

import FFI

data Event

alert :: String -&gt; Fay ()
alert = ffi &quot;alert(%1)&quot;

setBodyHtml :: String -&gt; Fay ()
setBodyHtml = ffi &quot;document.body.innerHTML = %1&quot;

addWindowEvent :: String -&gt; (Event -&gt; Fay ()) -&gt; Fay ()
addWindowEvent = ffi &quot;window.addEventListener(%1, %2)&quot;

greet :: Event -&gt; Fay ()
greet event = do
  putStrLn &quot;The document has loaded&quot;
  setBodyHtml &quot;Hello HTML!&quot;

main :: Fay ()
main = do
  putStrLn &quot;Hello Console!&quot;
  alert &quot;Hello Alert!&quot;
  addWindowEvent &quot;load&quot; greet
