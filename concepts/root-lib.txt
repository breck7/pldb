ROOT is an object-oriented program and library developed by CERN. It was originally designed for particle physics data analysis and contains several features specific to this field, but it is also used in other applications such as astronomy and data mining. #include &quot;Riostream.h&quot;
void basic() {
// read file $ROOTSYS/tutorials/tree/basic.dat
// this file has 3 columns of float data
   TString dir = gROOT-&gt;GetTutorialDir();
   dir.Append(&quot;/tree/&quot;);
   dir.ReplaceAll(&quot;/./&quot;,&quot;/&quot;);
   ifstream in;
   in.open(Form(&quot;%sbasic.dat&quot;,dir.Data()));
   Float_t x,y,z;
   Int_t nlines = 0;
   auto f = TFile::Open(&quot;basic.root&quot;,&quot;RECREATE&quot;);
   TH1F h1(&quot;h1&quot;,&quot;x distribution&quot;,100,-4,4);
   TNtuple ntuple(&quot;ntuple&quot;,&quot;data from ascii file&quot;,&quot;x:y:z&quot;);
   while (1) {
      in &gt;&gt; x &gt;&gt; y &gt;&gt; z;
      if (!in.good()) break;
      if (nlines &lt; 5) printf(&quot;x=%8f, y=%8f, z=%8f\n&quot;,x,y,z);
      h1.Fill(x);
      ntuple.Fill(x,y,z);
      nlines++;
   }
   printf(&quot; found %d points\n&quot;,nlines);
   in.close();
   f-&gt;Write();
}
