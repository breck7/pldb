seq range(start, end) {
    for (var i = start; i &lt;= end; i++)
        yield i;
}

seq where(list, filter) {
    foreach (var x in list) {
        if (filter(x))
            yield x;
    }
}

seq select(list, transform) {
    foreach (var x in list)
        yield transform(x);
}

fun toArray(list) {
    var array = [];

    foreach (var value in list) {
        array.add(value);
    }

    return array;
}

return range(0, 1000)
       |&gt; where(x -&gt; x % 2 == 0)
       |&gt; select(x -&gt; x / 2)
       |&gt; toArray();
