# dump stack from bottom to top
fn emit-stack-from-bottom _self: (addr grapheme-stack), out: (addr stream byte) {
  var self/esi: (addr grapheme-stack) &lt;- copy _self
  var data-ah/edi: (addr handle array code-point-utf8) &lt;- get self, data
  var _data/eax: (addr array code-point-utf8) &lt;- lookup *data-ah
  var data/edi: (addr array code-point-utf8) &lt;- copy _data
  var top-addr/ecx: (addr int) &lt;- get self, top
  var i/eax: int &lt;- copy 0
  {
    compare i, *top-addr
    break-if-&gt;=
    var g/edx: (addr code-point-utf8) &lt;- index data, i
    write-code-point-utf8 out, *g
    i &lt;- increment
    loop
  }
}
