{
	// Set timer of 1 second.
	SetTimer(&quot;OneSecTimer&quot;, 1000, 1);
	print(&quot;GameModeInit()&quot;);
	SetGameModeText(&quot;Timer Test&quot;);
	AddPlayerClass(0, 1958.3783, 1343.1572, 15.3746, 269.1425, 0, 0, 0, 0, 0, 0);
	return 1;
}

public OneSecTimer() {

	if(lasttick == 0) {
     	lasttick = GetTickCount();
		return;
	}
	new sText[256];
	format(sText,sizeof(sText),&quot;GetTickCountOffset = %d&quot;,GetTickCount() - lasttick);
	print(sText);
	SendClientMessageToAll(0xFF0000, sText);
	lasttick = GetTickCount();
}
