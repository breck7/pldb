# print fizzbuzz result
procedure fizzbuzz: .i
	if .i mod 15 == 0
		appendInfoLine: &quot;fizzbuzz&quot;
	elsif .i mod 3 == 0
		appendInfoLine: &quot;fizz&quot;
	elsif .i mod 5 == 0
		appendInfoLine: &quot;buzz&quot;
	else
		appendInfoLine: .i
	endif
endproc

for i from 1 to 100
	@fizzbuzz: i
endfor
