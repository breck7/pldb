MODULE      GCD.
IMPORT      Integers.
 
PREDICATE   Gcd : Integer * Integer * Integer.
Gcd(i,j,d) &lt;-
           CommonDivisor(i,j,d) &amp;
           ~ SOME [e] (CommonDivisor(i,j,e) &amp; e &gt; d).
 
PREDICATE   CommonDivisor : Integer * Integer * Integer.
CommonDivisor(i,j,d) &lt;-
           IF (i = 0 \/ j = 0)
           THEN
             d = Max(Abs(i),Abs(j))
           ELSE
             1 =&lt; d =&lt; Min(Abs(i),Abs(j)) &amp;
             i Mod d = 0 &amp;
             j Mod d = 0.
