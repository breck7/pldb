-- Raw temperature from sensor, range -50.0C to 100.0C.
temp :: Stream Word8
temp = extern &quot;temperature&quot; Nothing

-- Transform the temperature to Celsius.
ctemp :: Stream Float
ctemp = (unsafeCast temp) * (150.0 / 255.0) - 50.0

-- Bind two triggers with a 3.0C hysteresis.
spec = do
  trigger &quot;heaton&quot;  (ctemp &lt; 18.0) [arg ctemp]
  trigger &quot;heatoff&quot; (ctemp &gt; 21.0) [arg ctemp]
