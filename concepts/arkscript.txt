(import std.random)
(import std.Math)

(let number (mod (math:abs (random)) 10000))

(let game (fun () {
    (let impl (fun (tries) {
        (let guess (toNumber (input &quot;Input a numeric value: &quot;)))

        (if (&lt; guess number)
            {
                (print &quot;It&#39;s more than &quot; guess)
                (impl (+ tries 1))}
            (if (= guess number)
                {
                    (print &quot;You found it!&quot;)
                    tries }
                {
                    (print &quot;It&#39;s less than &quot; guess)
                    (impl (+ tries 1))}))}))

    (let tries (impl 0))
    (print &quot;You won in &quot; tries &quot; tries.&quot;)}))

(game)

(print &quot;Hello World&quot;)
