{
    # more or less game
    (print &quot;More or less game!&quot;)
    (import &quot;random.bin&quot;)
    (import &quot;Math/Arithmetic.ark&quot;)
    
    (let number (mod (abs (random)) 10000))
    (mut value 0)
    (mut essais 0)
    (mut continue true)
    
    (while continue {
        (set value (toNumber (input &quot;Input a numeric value: &quot;)))
        (if (&lt; value number)
            # then
            (print &quot;More!&quot;)
            # else
            (if (= value number)
                # then
                { (print &quot;Bingo!&quot;) (set continue false) }
                # else
                (print &quot;Less!&quot;)))
        (set essais (+ 1 essais))})
    (print &quot;You won in &quot; essais &quot; tries&quot;)
}
