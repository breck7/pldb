&#39; Specifies program will use functions from console module
uses &quot;Console&quot;

&#39; TBMain represents main body of the program
function TBMain()
  &#39; Creates variable to hold user name
  local UserName as string

  &#39; Asks user for the name
  Console_Print(&quot;What is your name?: &quot;)

  &#39; Stores it to variable
  UserName = Console_ReadLine

  &#39; If length of username is 0 then no name is specified, else program will say hello
  if len(UserName) = 0 then
    Console_PrintLine(&quot;No user name specified...&quot;)
  else
    Console_PrintLine(&quot;Hello &quot; + UserName + &quot;!&quot;)
  end if

  &#39; Waits for any key from user before program ends
  Console_WaitKey
end function
