some_user_defined_function_without_return_type(string inputAsString)
{
    llSay(PUBLIC_CHANNEL, inputAsString);
}

string user_defined_function_returning_a_string(key inputAsKey)
{
    return (string)inputAsKey;
}

default
{
    state_entry()
    {
        key someKey = NULL_KEY;
        someKey = llGetOwner();

        string someString = user_defined_function_returning_a_string(someKey);

        some_user_defined_function_without_return_type(someString);
    }

    touch_start(integer num_detected)
    {
        list agentsInRegion = llGetAgentList(AGENT_LIST_REGION, []);
        integer numOfAgents = llGetListLength(agentsInRegion);

        integer index;                                                          // defaults to 0
        for (; index &lt;= numOfAgents - 1; index++)                               // for each agent in region
        {
            llRegionSayTo(llList2Key(agentsInRegion, index), PUBLIC_CHANNEL, &quot;Hello, Avatar!&quot;);
        }
    }

    touch_end(integer num_detected)
    {
        someIntNormal       = 3672;
        someIntHex          = 0x00000000;
        someIntMath         = PI_BY_TWO;

        event               = 5673;// &#39;event&#39; is invalid.illegal

        someKeyTexture      = TEXTURE_DEFAULT;
        someStringSpecial   = EOF;

        llSetInventoryPermMask(&quot;some item&quot;, MASK_NEXT, PERM_ALL);// &#39;llSetInventoryPermMask&#39; is reserved.godmode

        llWhisper(PUBLIC_CHANNEL, &quot;Leaving \&quot;default\&quot; now...&quot;);
        state other;
    }
}

state other
{
    state_entry()
    {
        llWhisper(PUBLIC_CHANNEL, &quot;Entered \&quot;state other\&quot;, returning to \&quot;default\&quot; again...&quot;);
        state default;
    }
