# Contributor: Natanael Copa &lt;ncopa@alpinelinux.org&gt;

pkgname=abuild
pkgver=2.27.0
_ver=${pkgver%_git*}
pkgrel=0
pkgdesc=&quot;Script to build Alpine Packages&quot;
url=&quot;http://git.alpinelinux.org/cgit/abuild/&quot;
arch=&quot;all&quot;
license=&quot;GPL2&quot;
depends=&quot;fakeroot sudo pax-utils openssl apk-tools&gt;=2.0.7-r1 libc-utils
	attr tar pkgconf patch&quot;
if [ &quot;$CBUILD&quot; = &quot;$CHOST&quot; ]; then
	depends=&quot;$depends curl&quot;
fi
makedepends_build=&quot;pkgconfig&quot;
makedepends_host=&quot;openssl-dev&quot;
makedepends=&quot;$makedepends_host $makedepends_build&quot;
install=&quot;$pkgname.pre-install $pkgname.pre-upgrade&quot;
subpackages=&quot;apkbuild-cpan:cpan apkbuild-gem-resolver:gems&quot;
options=&quot;suid&quot;
pkggroups=&quot;abuild&quot;
source=&quot;http://dev.alpinelinux.org/archive/abuild/abuild-$_ver.tar.xz
	&quot;

_builddir=&quot;$srcdir/$pkgname-$_ver&quot;
prepare() {
	cd &quot;$_builddir&quot;
	for i in $source; do
		case $i in
		*.patch)
			msg &quot;Applying $i&quot;
			patch -p1 -i &quot;$srcdir&quot;/$i || return 1
			;;
		esac
	done
	sed -i -e &quot;/^CHOST=/s/=.*/=$CHOST/&quot; abuild.conf
}

build() {
	cd &quot;$_builddir&quot;
	make || return 1
}

package() {
	cd &quot;$_builddir&quot;
	make install DESTDIR=&quot;$pkgdir&quot; || return 1
	install -m 644 abuild.conf &quot;$pkgdir&quot;/etc/abuild.conf || return 1
	install -d -m 775 -g abuild &quot;$pkgdir&quot;/var/cache/distfiles || return 1
}

cpan() {
	pkgdesc=&quot;Script to generate perl APKBUILD from CPAN&quot;
	depends=&quot;perl perl-libwww perl-json&quot;
	arch=&quot;noarch&quot;
	mkdir -p &quot;$subpkgdir&quot;/usr/bin
	mv &quot;$pkgdir&quot;/usr/bin/apkbuild-cpan &quot;$subpkgdir&quot;/usr/bin/
}

gems() {
	pkgdesc=&quot;APKBUILD dependency resolver for RubyGems&quot;
	depends=&quot;ruby ruby-augeas&quot;
	arch=&quot;noarch&quot;
	mkdir -p &quot;$subpkgdir&quot;/usr/bin
	mv &quot;$pkgdir&quot;/usr/bin/apkbuild-gem-resolver &quot;$subpkgdir&quot;/usr/bin/
}
