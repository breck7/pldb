   counts = new Map()
   words = text.split(R&quot;\W+&quot;)
   words each word -&gt;
      current-count = counts.get(word) or 0
      counts.set(word, current-count + 1)
   consume(counts.entries()).sort(compare) where
      compare({w1, c1}, {w2, c2}) = c2 - c1
