-- recursive version, MANOOLish &quot;cascading&quot; notation
{ {extern &quot;manool.org.18/std/0.3/all&quot;} in
: let rec
  { Fact = -- compile-time constant binding
    { proc { N } as -- precondition: N.IsI48[] &amp; (N &gt;= 0)
    : if N == 0 then 1 else
      N * Fact[N - 1]
    }
  }
  in
  Out.WriteLine[&quot;Factorial of 10 is &quot;; Fact[10]]
}
