program ObjectPascalExample;

type
   THelloWorld = class
      procedure Put;
   end;

procedure THelloWorld.Put;
begin
   Writeln(&#39;Hello World&#39;);
end;

var
   HelloWorld: THelloWorld;

begin
   HelloWorld := THelloWorld.Create;
   HelloWorld.Put;
   HelloWorld.Free;
end.

type
   THelloWorld = class
      procedure Put;
      begin
         PrintLn(&#39;Hello, World!&#39;);
      end
   end;

var HelloWorld := THelloWorld.Create;

HelloWorld.Put;
