params.query = &quot;$HOME/sample.fa&quot;
params.db = &quot;$HOME/tools/blast-db/pdb/pdb&quot;

process blast {
    output:
     file top_hits

    &quot;&quot;&quot;
    blastp -query ${params.query} -db ${params.db} -outfmt 6 \
    | head -n 10 \
    | cut -f 2 &gt; top_hits
    &quot;&quot;&quot;
}

process extract {
    input:
     file top_hits
    output:
     file sequences

    &quot;&quot;&quot;
    blastdbcmd -db ${params.db} -entry_batch $top_hits &gt; sequences
    &quot;&quot;&quot;
}

process align {
    input:
     file sequences
    echo true

    &quot;&quot;&quot;
    t_coffee $sequences 2&gt;&amp;- | tee align_result
    &quot;&quot;&quot;
}
