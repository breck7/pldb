&#39;*************************************************************
&#39;** Set the configurable theme attributes for the application
&#39;**
&#39;** Configure the custom overhang and Logo attributes
&#39;** These attributes affect the branding of the application
&#39;** and are artwork, colors and offsets specific to the app
&#39;*************************************************************

Sub initTheme()
    app = CreateObject(&quot;roAppManager&quot;)
    app.SetTheme(CreateDefaultTheme())
End Sub

&#39;******************************************************
&#39;** @return The default application theme.
&#39;** Screens can make slight adjustments to the default
&#39;** theme by getting it from here and then overriding
&#39;** individual theme attributes.
&#39;******************************************************
Function CreateDefaultTheme() as Object
    theme = CreateObject(&quot;roAssociativeArray&quot;)

    theme.ThemeType = &quot;generic-dark&quot;

    theme.GridScreenMessageColor    = &quot;#808080&quot;

    theme.GridScreenListNameColor   = &quot;#FFFFFF&quot;
