  import config { prefix config; revision-date 2013-04-05; }
  import rpc-context { prefix rpcx; revision-date 2013-06-17; }
  import opendaylight-md-sal-binding { prefix mdsal; revision-date 2013-10-28; }

  description
      &quot;This module contains the base YANG definitions for
      sfc-lisp implementation.&quot;;

  revision &quot;2015-04-27&quot; {
      description
          &quot;Initial revision.&quot;;
  }

  // This is the definition of the service implementation as a module identity
  identity sfc-lisp-impl {
      base config:module-type;

      // Specifies the prefix for generated java classes.
      config:java-name-prefix SfcLisp;
  }

  // Augments the &#39;configuration&#39; choice node under modules/module.
  augment &quot;/config:modules/config:module/config:configuration&quot; {
    case sfc-lisp-impl {
      when &quot;/config:modules/config:module/config:type = &#39;sfc-lisp-impl&#39;&quot;;

      //wires in the data-broker service
      container data-broker {
        uses config:service-ref {
          refine type {
              mandatory false;
              config:required-identity mdsal:binding-async-data-broker;
          }
        }
      }

      container rpc-registry {
        uses config:service-ref {
          refine type {
              mandatory true;
              config:required-identity mdsal:binding-rpc-registry;
          }
        }
      }
    }
  }
}
