#include &quot;aldor&quot;
#include &quot;aldorio&quot;

import from Integer, String;

bob(n: Integer): String == {
    b: String := &quot; bottle&quot;;

    if n ~= 1 then b := b + &quot;s&quot;;
    b + &quot; of beer&quot;;
}

main(): () == {
    n: Integer := 99;
    otw: String := &quot; on the wall&quot;;

    -- refrain
    while n &gt; 0 repeat {
        stdout &lt;&lt; n &lt;&lt; bob(n) &lt;&lt; otw &lt;&lt; &quot;, &quot; &lt;&lt; n &lt;&lt; bob(n) &lt;&lt; &quot;.&quot; &lt;&lt; newline;
        stdout &lt;&lt; &quot;Take one down and pass it around, &quot;;
        n := n - 1;
        if n &gt; 0 then stdout &lt;&lt; n;
        else stdout &lt;&lt; &quot;no more&quot;;
        stdout &lt;&lt; bob(n) &lt;&lt; otw &lt;&lt; &quot;.&quot; &lt;&lt; newline;
        stdout &lt;&lt; newline;
    }

    -- last verse
    stdout &lt;&lt; &quot;No more&quot; &lt;&lt; bob(n) &lt;&lt; otw &lt;&lt; &quot;, no more&quot; &lt;&lt; bob(n) &lt;&lt; &quot;.&quot; &lt;&lt; newline;
    stdout &lt;&lt; &quot;Go to the store and buy some more, &quot;;
    n: Integer := 99;
    stdout &lt;&lt; n &lt;&lt; bob(n) &lt;&lt; otw &lt;&lt; &quot;.&quot; &lt;&lt; newline;
}

main();
