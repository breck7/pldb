package Sorting api;

fn Partition[T:! Comparable &amp; Movable](s: Slice(T))
     -&gt; i64 {
  var i: i64 = -1;

  for (e: T in s) {
    if (e &lt;= s.Last()) {
      ++i;
      Swap(&amp;s[i], &amp;e);
    }
  }
  return i;
}

fn QuickSort[T:! Comparable &amp; Movable](s: Slice(T)) {
  if (s.Size() &lt;= 1) {
    return;
  }
  let p: i64 = Partition(s);
  QuickSort(s[:p - 1]));
  QuickSort(s[p + 1:]));
}
