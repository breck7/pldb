IF .NOT. FILE(dbname + &quot;.dbf&quot;)                  // database available?
  DBCREATE(dbname, {{&quot;Name&quot;,    &quot;C&quot;, 25, 0}, ;  //  not yet, so
                    {&quot;Address&quot;, &quot;C&quot;, 30, 0}, ;  //  create a new
                    {&quot;Born&quot;,    &quot;D&quot;,  8, 0}, ;  //  database, here
                    {&quot;Note&quot;,    &quot;M&quot;, 10, 0}} )  //  with 4 fields
ENDIF
USE (dbname) SHARED         // open the given database or mydbf1.dbf
IF !USED()                  // check the success
   ? &quot;sorry, cannot open&quot;, dbname
   QUIT                     // abort the execution
ENDIF
BROWSE ()                   // browse/edit
QUIT                        // exit back to command line
