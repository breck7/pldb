/**
 * Says hello to the world.
 */
{template .helloWorld}
  Hello world!
{/template}

{namespace Exmaple}

/**
 * Example
 */
{template .foo}
  {@param count: string}
  {@param? name: int}

  {if isNonnull($name)}
    &lt;h1&gt;{$name}&lt;/h1&gt;
  {/if}

  &lt;div class=&quot;content&quot;&gt;
    {switch count}
      {case 0}
        {call Empty.view}
          {param count: $count /}
        {/call}
      {default}
        &lt;h2&gt;Wow, so many!&lt;/h2&gt;
    {/switch}
  &lt;/div&gt;
{/template}
