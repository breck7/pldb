use fmt;

export fn main() void = {
	const greetings = [
		&quot;Hello, world!&quot;,
		&quot;¡Hola Mundo!&quot;,
		&quot;Γειά σου Κόσμε!&quot;,
		&quot;Привет, мир!&quot;,
		&quot;こんにちは世界！&quot;,
	];
	for (let i = 0z; i &lt; len(greetings); i += 1) {
		fmt::println(greetings[i])!;
	};
};
