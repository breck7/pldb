Manual Memory Management
========================

#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
int main(void)
{
  int *poin = malloc(4);
  free(poin);
}

use std::mem::c::{malloc, free}

fn main() {
    let mut ptr = malloc(8)
    free(ptr)
    ptr = nil
}
